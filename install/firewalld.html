Centos7

дәҢгҖҒе®үиЈ… вҖӢ

shell
$ yum install -y firewalld

дёүгҖҒ еҹәжң¬еҗҜеҠЁе‘Ҫд»Ө вҖӢ

shell
$ systemctl status firewalld                # жҹҘзңӢзҠ¶жҖҒ
$ systemctl start firewalld                  # еҗҜеҠЁ
$ systemctl stop firewalld                  #е…ій—ӯ
$ systemctl enable firewalld              # ејҖжңәеҗҜеҠЁ
$ systemctl disable firewalld             # еҸ–ж¶ҲејҖжңәеҗҜеҠЁ

еӣӣгҖҒеёёз”Ёе‘Ҫд»Ө вҖӢ

1. жҹҘзңӢ вҖӢ

shell
$ firewall-cmd --get-active-zones   # жҹҘзңӢжҝҖжҙ»зҡ„еҹҹ
$ firewall-cmd --zone=public --list-ports  # жҹҘзңӢејҖж”ҫзҡ„з«ҜеҸЈ
$ firewall-cmd --zone=public --list-rich-rules  # жҹҘзңӢж·»еҠ зҡ„и§„еҲҷ

2. ж·»еҠ з«ҜеҸЈ вҖӢ

shell
# ејҖж”ҫеҚ•дёӘз«ҜеҸЈ
$ firewall-cmd --zone=public --add-port=80/tcp --permanent                     

# ејҖж”ҫз«ҜеҸЈиҢғеӣҙ
$ firewall-cmd --zone=public --add-port=8388-8389/tcp --permanent        

# еҜ№ 147.152.139.197 ејҖж”ҫ10000з«ҜеҸЈ
$ firewall-cmd --permanent --zone=public --add-rich-rule='
        rule family="ipv4"
        source address="147.152.139.197/32"
        port protocol="tcp" port="10000" accept'       
        
# жӢ’з»қз«ҜеҸЈпјҡ
$ firewall-cmd --permanent --zone=public --add-rich-rule='
              rule family="ipv4"
              source address="47.52.39.197/32"
              port protocol="tcp" port="10000" reject'  

# ејҖж”ҫе…ЁйғЁз«ҜеҸЈз»ҷIP
$ firewall-cmd --permanent --zone=public --add-rich-rule='
              rule family="ipv4"
              source address="192.168.0.1/32" accept';

# ејҖж”ҫе…ЁйғЁз«ҜеҸЈз»ҷзҪ‘ж®ө
$ firewall-cmd --permanent --zone=public --add-rich-rule='
              rule family="ipv4"
              source address="192.168.0.0/16" accept';

3. ж·»еҠ жңҚеҠЎ вҖӢ

shell
# жҹҘзңӢе…ЁйғЁж”ҜжҢҒзҡ„жңҚеҠЎ
$ firewall-cmd --get-service

# жҹҘзңӢејҖж”ҫзҡ„жңҚеҠЎ
$ firewall-cmd --list-service

# ж·»еҠ жңҚеҠЎ,ж·»еҠ https
$ firewall-cmd --add-service=https --permanent

дҝ®ж”№еҜ№еә”зҡ„й…ҚзҪ®ж–Үд»¶жҳҜ/etc/firewalld/zones/public.xml

4. з§»йҷӨз«ҜеҸЈ вҖӢ

shell
# з§»йҷӨж·»еҠ зҡ„з«ҜеҸЈ
$ firewall-cmd --zone=public --remove-port=80/tcp --permanent

4. йҮҚиҪҪй…ҚзҪ® вҖӢ

еҜ№и·Ҝз”ұи§„еҲҷиҝӣиЎҢдҝ®ж”№еҗҺпјҢйңҖиҰҒйҮҚж–°еҠ иҪҪи§„еҲҷжүҚиғҪдҪҝи§„еҲҷз”ҹж•Ҳ

shell
$ firewall-cmd --reload
